React Integration (Code Example)

// React component using ESM web components
function MyReactComponent() {
  const [model, setModel] = useState(initialModel);

  return (
    <esm-model-editor
      model={JSON.stringify(model)}
      allow-editing="true"
      onModelChange={(e) => setModel(e.detail)}
    />
  );
}
