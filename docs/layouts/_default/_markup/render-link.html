{{- $url := .Destination -}} {{- $isAbs := (urls.Parse $url).IsAbs -}} {{- if not $isAbs -}} {{- $frag := "" -}} {{- $path := $url -}} {{- if in $url "#" -}} {{- $parts := split $url "#" -}} {{- $path = index $parts 0 -}} {{- $frag = index $parts 1 -}} {{- end -}} {{- $page := .Page.GetPage $path -}} {{- if $page -}} {{- $url = $page.RelPermalink -}} {{- if ne $frag "" -}} {{- $url = printf "%s#%s" $url $frag -}} {{- end -}} {{- else if or (strings.HasSuffix $path ".md") (in $url ".md#") -}} {{- $url = replace $url ".md" "" 1 -}} {{- end -}} {{- end -}} {{ .Text | safeHTML }}
